\begin{frame}
\frametitle{How to Configure and Build}
\framesubtitle{cmake-gui}
\begin{itemize}
\item Create a directory
\item Configure the code with CMake
\item Build (compile and link an executable)
\item Run it in example image
\end{itemize}
\end{frame}
